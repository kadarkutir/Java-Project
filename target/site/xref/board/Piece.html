1   package board;

3   import javafx.beans.property.ObjectProperty;
4   import javafx.beans.property.SimpleObjectProperty;
5   
6   /**
7    * Represents a piece
8    */
9   public class Piece {
10  
11      /**
12       * The type of the piece.
13       */
14      private final PieceType type;
15  
16      /**
17       * The position of the piece.
18       */
19      private final ObjectProperty<Position> position = new SimpleObjectProperty<>();
20  
21      public Piece(PieceType type, Position position) {
22          this.type = type;
23          this.position.set(position);
24      }
25  
26      /**
27       *{@return the type of the piece}
28       */
29      public PieceType getType() { return type; }
30  
31      /**
32       *{@return the position of the piece}
33       */
34      public Position getPosition() {
35          return position.get();
36      }
37  
38      public ObjectProperty<Position> positionProperty() {
39          return position;
40      }
41  
42      /**
43       * Moves the piece to a given direction.
44       *
45       * @param direction a given direction
46       */
47      public void moveTo(Direction direction) {
48          Position newPosition = position.get().moveTo(direction);
49          position.set(newPosition);
50      }
51  
52      @Override
53      public String toString(){ return type.toString() + " " + position; }
54  
55  }
